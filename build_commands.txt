#Linux
pyinstaller --onefile --name voxcall --noconfirm --clean \
  --add-data "voxcall/resources:resources" \
  --add-data "voxcall/webui/templates:voxcall/webui/templates" \
  --add-data "voxcall/webui/static:voxcall/webui/static" \
  --collect-submodules PIL \
  --collect-data PIL \
  --hidden-import PIL._imagingtk \
  --hidden-import PIL._tkinter_finder \
  --collect-submodules ttkbootstrap \
  voxcall.py

mkdir -p ~/.local/share/icons/hicolor/256x256/apps
cp /path/to/voxcall.png ~/.local/share/icons/hicolor/256x256/apps/voxcall.png
update-desktop-database ~/.local/share/applications >/dev/null 2>&1 || true

[Desktop Entry]
Type=Application
Name=VoxCall
Exec=/full/path/to/voxcall
Icon=voxcall
Terminal=false
Categories=AudioVideo;Utility;
StartupWMClass=VoxCall
